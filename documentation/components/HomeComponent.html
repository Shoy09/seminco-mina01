})

  usuario: string = 'Jhoel Dioses!';
  mostrarConfiguraciones: boolean = true;
  mostrarAlerta: boolean = true;

  toggleConfiguraciones() {
    this.mostrarConfiguraciones = !this.mostrarConfiguraciones;
  }

  closeAlert() {
    this.mostrarAlerta = false;
  }
}
